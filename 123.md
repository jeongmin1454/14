### todo list

todo_list = [] 

while True:
    print("[ 메뉴: 1.추가  2.보기  3.완료체크  4.삭제  5.종료 ]")
    option = input("번호를 입력하세요: ")

    if option == "1":
        work = input("목표를 입력하세요:")
        todo_list.append(work)
        print(f"{work}를 추가했습니다.")

    elif option == "2":
        print("오늘의 할일")
        if len(todo_list) <= 0:
            print("할일을 추가하세요.")
        else:
            for i in range(len(todo_list)):
                print(f"{i+1}. {todo_list[i]}")

    elif option == "3":
        if len(todo_list) <= 0:
            print("완료할 목록이 없습니다.")
        num = int(input("완료한 번호"))
        real_num = num - 1     #인덱스 번호로 저장하기 위해

        if "(완료)" not in todo_list[real_num]:
            todo_list[real_num] += " (완료)"
            print(f"{num}번이 완료되었습니다.")
        elif "(완료)" in todo_list[real_num]:
            print("이미 완료했습니다")
        else:
            print("목록을 확인하세요.")

    elif option == "4":
        num = int(input("삭제할 번호"))
        real_num = num - 1     #인덱스 번호로 저장하기 위해
        if len(todo_list) <= 0:
            print("목록이 없습니다")
        elif len(todo_list) > 0:
            todo_list.pop(real_num)
            print(f"{num}번이 삭제되었습니다.")
        else:
            print("목록을 확인하세요")

    elif option == "5":
        print("todo list 종료")
        break

    else:
        print("1~5까지만 입력하세요")
